int max;
max = 100000;

int i;
i = 2;

bool found;
int count;
int temp;

int prev;

while(i <= max){
	if(i%2!=0 && i%5!=0){
		temp = (i+1)/2;
		count = 2;
		found = true;
		while(count<=temp){
			if(i%count==0){
				found = false;
			}
			count = count + 1;
		}
		if(found){
			if(i-prev==2){
				print(prev);
				print(i);
				print(true);
			}
			prev = i;
		}
	}
	i = i + 1;
}
