int max;
int i;
int tracking;
bool found;

max = 1000;

i = 2;
while(i <= max){
	if((i%2!=0 && i%5!=0) || i<7){
		tracking = 2;
		found = true;
		while(tracking < (i/2)){
			if(i % tracking == 0){
				found = false;
			}
			tracking = tracking + 1;
		}
		if(found){
			print(i);
		}
	}
	i = i + 1;
}
